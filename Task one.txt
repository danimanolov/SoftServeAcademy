The N'th digit of N*N number

using System;
using System.Numerics;
using System.Text;

namespace GitHub_All_Homeworks_Done
{
    class Program
    {
        static void Main(string[] args)
        {
            int n = int.Parse(Console.ReadLine());
            string number = Math.Pow(n,n).ToString();
            string number2 = numberPow(n);
            char numberAtPositionN = number.ToCharArray()[n];
            char numberAtPositionN2 = number2.ToCharArray()[n];
            Console.WriteLine(numberAtPositionN);
            Console.WriteLine(numberAtPositionN2);
        }
        static string numberPow(int n)
        {
            BigInteger result = 1;
            int temp = n;
            while(temp > 0)
            {
                result *= n;
                temp--;
            }

            return result.ToString();
        }
    }
}
